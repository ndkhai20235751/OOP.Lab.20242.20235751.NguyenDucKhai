package hello.java;

import java.util.Scanner; // Import thư viện Scanner

public class hello { // Tên class nên viết hoa chữ cái đầu
    public static void main(String[] args) {
        Scanner input = new Scanner(System.in); // Sử dụng tên biến đúng

        System.out.print("Nhập số thứ nhất: ");
        double num1 = Double.parseDouble(input.nextLine()); // Đổi từ scanner -> input

        System.out.print("Nhập số thứ hai: ");
        double num2 = Double.parseDouble(input.nextLine()); // Đổi từ scanner -> input

        double sum = num1 + num2;
        double difference = num1 - num2;
        double product = num1 * num2;

        if (num2 != 0) {
            double quotient = num1 / num2;
            System.out.println("Thương: " + quotient);
        } else {
            System.out.println("Không thể chia cho 0.");
        }

        System.out.println("Tổng: " + sum);
        System.out.println("Chênh lệch: " + difference);
        System.out.println("Tích: " + product);

        input.close(); // Đóng Scanner
    }
}
